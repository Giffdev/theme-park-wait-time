## Firestore Schema

```typescript
interface ParkDaySchedule {
  parkId: string;
  date: string;
  timezone: string;
  segments: Array<{
    type: 'OPERATING' | 'TICKETED_EVENT' | 'EXTRA_HOURS';
    description: string | null;
    openingTime: string;
    closingTime: string;
    purchases?: Array<{
      name: string;
      price: { amount, currency, formatted };
      available: boolean;
    }>;
  }>;
  fetchedAt: string;
}
```
